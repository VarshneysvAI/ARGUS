import streamlit as st
from fpdf import FPDF
import plotly.io as pio
from datetime import datetime

class ARGUSReport(FPDF):
    def header(self):
        self.set_font('Arial', 'B', 16)
        self.set_text_color(231, 76, 60)  # #e74c3c
        self.cell(0, 10, 'ARGUS Energy Supply Chain Risk Report', 0, 1, 'C')
        self.set_font('Arial', '', 10)
        self.set_text_color(136, 153, 170)
        self.cell(0, 5, f'Generated: {datetime.now().strftime("%Y-%m-%d %H:%M UTC")}', 0, 1, 'C')
        self.ln(5)
    
    def chapter_title(self, title):
        self.set_font('Arial', 'B', 12)
        self.set_text_color(52, 152, 219)  # #3498db
        self.cell(0, 8, title, 0, 1, 'L')
        self.ln(2)
    
    def chapter_body(self, body):
        self.set_font('Arial', '', 10)
        self.set_text_color(255, 255, 255)
        self.multi_cell(0, 5, body)
        self.ln()

def generate_pdf(risk_score, risk_level, confidence, agents_data, recommendations):
    pdf = ARGUSReport()
    pdf.add_page()
    pdf.set_fill_color(5, 8, 16)  # Dark background
    pdf.rect(0, 0, 210, 297, 'F')
    
    # Executive Summary
    pdf.chapter_title('EXECUTIVE SUMMARY')
    pdf.chapter_body(f"""
Risk Score: {risk_score} ({risk_level})
System Confidence: {confidence}
Incident: Strait of Hormuz Closure
Commodity: Crude Oil
Economy: India
    """)
    
    # Agent Outputs
    pdf.chapter_title('AGENT ANALYSIS')
    for agent in agents_data:
        pdf.chapter_body(f"""
Agent {agent['id']}: {agent['name']}
Output: {agent['output']}
Confidence: {agent['confidence']}
Sources: {', '.join(agent['sources'])}
        """)
    
    # Recommendations
    pdf.chapter_title('ALTERNATIVE SOURCING RECOMMENDATIONS')
    for rec in recommendations:
        pdf.chapter_body(f"""
Rank {rec['rank']}: {rec['route']}
Score: {rec['score']}
Cost Delta: {rec['cost_delta']}
Lead Time: {rec['lead_time']} days
Source: {rec['source_url']}
        """)
    
    # Uncertainty Disclosure
    pdf.chapter_title('UNCERTAINTY DISCLOSURE')
    pdf.chapter_body("""
This report is based on:
- Verified government data (EIA/IEA/PIB): 73%
- Inferred from historical patterns: 17%
- LLM narrative synthesis: 10%

Known gaps:
- Real-time AIS data not integrated
- Refinery blending costs not modeled
- Geopolitical negotiation outcomes unpredictable
    """)
    
    # Save
    filename = f"ARGUS_Report_{datetime.now().strftime('%Y%m%d_%H%M')}.pdf"
    pdf.output(filename)
    return filename

# Agent 1D: Image & Map Retriever
class ImageRetriever:
    def __init__(self):
        self.ais_stream = AISstreamClient()  # Free WebSocket
        self.sentinel = SentinelAPI()  # Free Copernicus
    
    def retrieve(self, corridor, commodity):
        images = []
        
        # 1. Live vessel positions from AISstream.io
        vessels = self.ais_stream.get_vessels_in_bbox(
            bbox=self.corridor_to_bbox(corridor)  # Hormuz bounding box
        )
        images.append({
            "type": "ais_map",
            "data": vessels,
            "source": "AISstream.io",
            "url": "wss://stream.aisstream.io/..."
        })
        
        # 2. Satellite imagery from Sentinel-2
        sat_img = self.sentinel.query(
            footprint=self.corridor_to_footprint(corridor),
            date=("20260301", "20260331"),  # March 2026 crisis
            platformname="Sentinel-2"
        )
        images.append({
            "type": "satellite",
            "data": sat_img,
            "source": "Copernicus Sentinel-2",
            "url": "https://scihub.copernicus.eu/..."
        })
        
        # 3. Port congestion from MarineTraffic (free tier)
        congestion = self.scrape_marinetraffic(corridor)
        images.append({
            "type": "congestion",
            "data": congestion,
            "source": "MarineTraffic",
            "url": "https://www.marinetraffic.com/..."
        })
        
        return images
